FROM python:3.11-slim

WORKDIR /app


# =========================
# 系統依賴（SSL 修復關鍵）
# =========================
RUN apt-get update && apt-get install -y \
    ca-certificates \
    curl \
    && update-ca-certificates \
    && rm -rf /var/lib/apt/lists/*


# =========================
# Python dependencies
# =========================
COPY requirements.txt .

RUN pip install --no-cache-dir -r requirements.txt


# =========================
# copy project
# =========================
COPY . .


# =========================
# port
# =========================
EXPOSE 10000


# =========================
# production run
# =========================
CMD ["gunicorn", \
     "-w", "2", \
     "-b", "0.0.0.0:10000", \
     "--timeout", "120", \
     "app:app"]
